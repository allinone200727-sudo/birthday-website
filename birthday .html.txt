        document.querySelector('.pulsating-dot').addEventListener('click', () => {
            document.querySelector('.landing').style.display = 'none';
            const sections = document.querySelectorAll('.reveal, .gallery, .reasons, .message');
            sections.forEach(section => section.style.display = 'block');
        });

        const reasonItems = document.querySelectorAll('.reasons li');
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = 1;
                    entry.target.style.transform = 'translateX(0)';
                }
            });
        }, { threshold: 0.5 });

        reasonItems.forEach((item, index) => {
            item.style.transitionDelay = `${index * 0.2}s`;
            observer.observe(item);
        });
